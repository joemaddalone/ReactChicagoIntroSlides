Use React as the UI for other frameworks

app.directive('reactDirective', function(){
  return {
    restrict: 'E',
    scope: {data:'='},
    link: function( scope, elm, attr ){
      scope.$watch( 'data', function( newVal, oldVal ){
        React.render( APP( { data: newVal } ), elm[0]);
      })
    }
  }
})
